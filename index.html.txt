    function nowStamp() {
      const d = new Date();
      const pad = n => String(n).padStart(2, '0');
      return `${d.getFullYear()}-${pad(d.getMonth()+1)}-${pad(d.getDate())} ${pad(d.getHours())}:${pad(d.getMinutes())}:${pad(d.getSeconds())}`;
    }

    function toNumber(value) {
      const n = Number(String(value ?? '').replace(',', '.').trim());
      return isNaN(n) ? 0 : n;
    }

    function formatNumber(value) {
      const n = toNumber(value);
      return Number.isInteger(n) ? String(n) : String(n);
    }

    function escapeHtml(value) {
      return String(value ?? '')
        .replaceAll('&', '&amp;')
        .replaceAll('<', '&lt;')
        .replaceAll('>', '&gt;')
        .replaceAll('"', '&quot;')
        .replaceAll("'", '&#039;');
    }

    function normalizeCode(value) {
      return String(value ?? '').trim();
    }

    function rebuildFilters() {
      const current = $('categoryFilter').value;
      const categories = [...new Set(originalData.map(x => x['Categorie']).filter(Boolean))].sort((a,b) => String(a).localeCompare(String(b), 'ro'));
      $('categoryFilter').innerHTML = '<option value="">Toate categoriile</option>';
      for (const cat of categories) {
        const option = document.createElement('option');
        option.value = cat;
        option.textContent = cat;
        $('categoryFilter').appendChild(option);
      }
      if (categories.includes(current)) $('categoryFilter').value = current;
    }

    function recalcRow(row) {
      row['Stoc initial'] = formatNumber(row['Stoc initial']);
      row['Intrari'] = formatNumber(row['Intrari']);
      row['Iesiri'] = formatNumber(row['Iesiri']);
      row['Stoc final'] = formatNumber(toNumber(row['Stoc initial']) + toNumber(row['Intrari']) - toNumber(row['Iesiri']));
      row['Categorie'] = String(row['Categorie'] ?? '');
      row['Cod produs'] = normalizeCode(row['Cod produs']);
      row['Denumire produs'] = String(row['Denumire produs'] ?? '');
      row['UM'] = String(row['UM'] ?? '');
      row['Data intrari'] = String(row['Data intrari'] ?? '');
      row['Data iesiri'] = String(row['Data iesiri'] ?? '');
    }

    function applyFilters() {
      const q = $('search').value.trim().toLowerCase();
      const category = $('categoryFilter').value;
      const stock = $('stockFilter').value;

      filteredData = originalData.filter(row => {
        const matchesSearch = !q || Object.values(row).some(v => String(v).toLowerCase().includes(q));
        const matchesCategory = !category || row['Categorie'] === category;
        const stockValue = toNumber(row['Stoc final']);
        const matchesStock =
          !stock ||
          (stock === 'positive' && stockValue > 0) ||
          (stock === 'zero' && stockValue === 0) ||
          (stock === 'negative' && stockValue < 0);
        return matchesSearch && matchesCategory && matchesStock;
      });

      if (sortKey) filteredData.sort(compareRows(sortKey, sortDirection));
      currentPage = 1;
      renderTable();
      updateStats();
    }

    function compareRows(key, direction) {
      return (a, b) => {
        const av = a[key] ?? '';
        const bv = b[key] ?? '';
        const aNum = toNumber(av);
        const bNum = toNumber(bv);
        const bothNumeric = String(av).trim() !== '' && String(bv).trim() !== '' && !isNaN(aNum) && !isNaN(bNum);
        let result = bothNumeric
          ? aNum - bNum
          : String(av).localeCompare(String(bv), 'ro', { numeric: true, sensitivity: 'base' });
        return direction === 'asc' ? result : -result;
      };
    }

    function sortData(key) {
      if (sortKey === key) {
        sortDirection = sortDirection === 'asc' ? 'desc' : 'asc';
      } else {
        sortKey = key;
        sortDirection = 'asc';
      }
      filteredData.sort(compareRows(key, sortDirection));
      renderTable();
    }

    function changeEntry(index, delta) {
      const row = originalData[index];
      row['Intrari'] = formatNumber(toNumber(row['Intrari']) + delta);
      row['Data intrari'] = nowStamp();

      saveData();
      applyFilters();
    }

    function changeExit(index, delta) {
      const row = originalData[index];
      row['Iesiri'] = formatNumber(toNumber(row['Iesiri']) + delta);
      row['Data iesiri'] = nowStamp();
      recalcRow(row);
      saveData();
      applyFilters();
    }

    function updateField(index, field, value) {
      originalData[index][field] = value;
      if (field === 'Intrari' || field === 'Iesiri' || field === 'Stoc initial') {
        if (field === 'Intrari') originalData[index]['Data intrari'] = nowStamp();
        if (field === 'Iesiri') originalData[index]['Data iesiri'] = nowStamp();
      }
      recalcRow(originalData[index]);
      saveData();
      applyFilters();
    }

    function deleteProduct(index) {
      const row = originalData[index];
      const ok = confirm(`Ștergi produsul ${row['Cod produs']} - ${row['Denumire produs']} ?`);
      if (!ok) return;
      originalData.splice(index, 1);
      saveData();
      rebuildFilters();
      applyFilters();
    }

    function generateNextCode() {
      const prefix = 'INV';
      const numericParts = originalData
        .map(x => String(x['Cod produs'] || '').match(/^INV(\d+)$/i))
        .filter(Boolean)
        .map(m => Number(m[1]));
      const next = (numericParts.length ? Math.max(...numericParts) : 0) + 1;
      return prefix + String(next).padStart(6, '0');
    }

    function renderCode128SVG(text, width = 220, height = 72) {
      text = normalizeCode(text);
      if (!text) return `<svg xmlns="http://www.w3.org/2000/svg" width="${width}" height="${height}"><text x="10" y="30">Fără cod</text></svg>`;

      // Code128 subset B
      const patterns = [
        "11011001100","11001101100","11001100110","10010011000","10010001100","10001001100","10011001000","10011000100",
        "10001100100","11001001000","11001000100","11000100100","10110011100","10011011100","10011001110","10111001100",

        "11000101000","11000100010","10110111000","10110001110","10001101110","10111011000","10111000110","10001110110",
        "11101110110","11010001110","11000101110","11011101000","11011100010","11011101110","11101011000","11101000110",
        "11100010110","11101101000","11101100010","11100011010","11101111010","11001000010","11110001010","10100110000",
        "10100001100","10010110000","10010000110","10000101100","10000100110","10110010000","10110000100","10011010000",
        "10011000010","10000110100","10000110010","11000010010","11001010000","11110111010","11000010100","10001111010",
        "10100111100","10010111100","10010011110","10111100100","10011110100","10011110010","11110100100","11110010100",
        "11110010010","11011011110","11011110110","11110110110","10101111000","10100011110","10001011110","10111101000",
        "10111100010","11110101000","11110100010","10111011110","10111101110","11101011110","11110101110","11010000100",
        "11010010000","11010011100","1100011101011"
      ];

      const START_B = 104;
      const STOP = 106;

      const codes = [START_B];
      for (let i = 0; i < text.length; i++) {
        const code = text.charCodeAt(i) - 32;
        if (code < 0 || code > 95) {
          return `<svg xmlns="http://www.w3.org/2000/svg" width="${width}" height="${height}">
            <rect width="100%" height="100%" fill="white"/>
            <text x="10" y="24" font-size="14" fill="red">Cod invalid pentru Code128</text>
            <text x="10" y="46" font-size="14">${text.replace(/&/g,'&amp;').replace(/</g,'&lt;')}</text>
          </svg>`;
        }
        codes.push(code);
      }

      let checksum = START_B;
      for (let i = 0; i < text.length; i++) checksum += (text.charCodeAt(i) - 32) * (i + 1);
      checksum = checksum % 103;
      codes.push(checksum, STOP);

      const bits = codes.map(c => patterns[c]).join('');
      const quiet = 10;
      const moduleWidth = Math.max(1, Math.floor((width - quiet * 2) / bits.length));
      const actualWidth = bits.length * moduleWidth + quiet * 2;
      const barHeight = height - 18;

      let x = quiet;
      let rects = '';
      for (const bit of bits) {
        if (bit === '1') rects += `<rect x="${x}" y="0" width="${moduleWidth}" height="${barHeight}" fill="black"/>`;
        x += moduleWidth;
      }

      const safeText = text.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');
      return `<svg xmlns="http://www.w3.org/2000/svg" width="${actualWidth}" height="${height}" viewBox="0 0 ${actualWidth} ${height}">
        <rect width="100%" height="100%" fill="white"/>
        ${rects}
        <text x="${actualWidth/2}" y="${height-2}" text-anchor="middle" font-family="Arial" font-size="14">${safeText}</text>
      </svg>`;
    }

      const exact = originalData.find(row => String(row['Cod produs']).trim().toLowerCase() === normalized);
      if (exact) {
        $('scanMessage').innerHTML = `<span class="success">Produs găsit:</span> ${escapeHtml(exact['Cod produs'])} - ${escapeHtml(exact['Denumire produs'])}`;
      } else {
        $('scanMessage').innerHTML = `<span class="danger">Nu am găsit codul:</span> ${escapeHtml(code)}`;
      }
    }

    async function startCameraScanner() {
      if (!navigator.mediaDevices || !navigator.mediaDevices.getUserMedia) {
        $('scanMessage').innerHTML = '<span class="danger">Browserul nu permite accesul la cameră.</span>';
        return;
      }

      try {
        stream = await navigator.mediaDevices.getUserMedia({
          video: { facingMode: { ideal: 'environment' } },
          audio: false
        });
        $('video').srcObject = stream;
        $('videoWrap').style.display = 'block';
        $('scanMessage').innerHTML = 'Camera este pornită. Ține codul de bare în fața camerei.';

        if ('BarcodeDetector' in window) {
          const detector = new BarcodeDetector({
            formats: ['ean_13','ean_8','code_128','code_39','qr_code','upc_a','upc_e']
          });

          if (scanTimer) clearInterval(scanTimer);
          scanTimer = setInterval(async () => {
            try {
              const barcodes = await detector.detect($('video'));
              if (barcodes && barcodes.length) {
                const code = barcodes[0].rawValue;
                $('scanInput').value = code;
                focusFoundProduct(code);
              }
            } catch (e) {}
          }, 800);
        } else {
          $('scanMessage').innerHTML = 'Camera este pornită. Browserul nu are scanare automată; poți folosi scanner extern sau introducere manuală.';
        }
      } catch (e) {
        $('scanMessage').innerHTML = '<span class="danger">Nu s-a putut porni camera. Verifică permisiunea de cameră.</span>';
      }
    }

    function stopCameraScanner() {
      if (scanTimer) {
        clearInterval(scanTimer);
        scanTimer = null;
      }
      if (stream) {
        stream.getTracks().forEach(t => t.stop());
        stream = null;
      }
      $('videoWrap').style.display = 'none';
      $('scanMessage').textContent = 'Camera este oprită.';
    }
